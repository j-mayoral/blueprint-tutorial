\begin{definition}[Sum until]
  \label{def:sum_until}
  \lean{sum_until}
  For every natural number $n \in \mathbb{N}$, we define
  \[
    \mathrm{sum\_until}(n) = \sum_{k = 0}^{n} k.
  \]
\end{definition}

\begin{theorem}[Sum of the first $n$ natural numbers]
  \label{thm:sum_nats}
  \lean{sum_first_natural_numbers}
  \leanok
  \uses{def:sum_until}
  For every $n \in \mathbb{N}$,
  \[
    \mathrm{sum\_until}(n) = \frac{n(n+1)}{2}.
  \]
\end{theorem}

\begin{proof}
  \leanok
  This follows from a straightforward induction on $n$.
\end{proof}
